\plush{
  \pptBanner{When AI Is Uncertain, It Asks Me}
  \begin{itemize}
  \item something is missing in the "intent"
  \item most often it asks about design, not implementation details
  \item it understands the implication
  \item it expects me to fill the \textbf{gaps}
  \end{itemize}
  }

\plush{
  \pptBanner{We, Humans, Are the Integrators}
  \begin{itemize}
  \item responsibility is ours, we understand the consequences
  \item we put pieces together and delegate implementation to AI
  \item bottleneck was never code --- it was holding the system in our head
  \item AI now writes; we still hold --- that's what \textbf{burns us out}
  \end{itemize}
  }

\plush{
  \pptBanner{AI Holds the System --- We Hold the Intent}
  \begin{itemize}
  \item every abstraction ate the one below, since Assembly
  \item SSH, SQL, IDEs, editors --- the next layer to disappear
  \item AI will know it all: software, data, servers, network, hardware
  \item we will only tell it what we \textbf{need}
  \end{itemize}
  }

\plush{
  \pptBanner{Then the Machine Asks Us}
  \begin{itemize}
  \item the system is text --- AI reads it
  \item the org, the politics, the user's frustration --- not text
  \item AI must ask: go there, ask that person, come back with the answer
  \item AI will know how systems work --- we will \textbf{feel them}
  \end{itemize}
  }

\plush{
  \pptBanner{We Become the Senses}
  \begin{itemize}
  \item we become concern providers --- AI's senses: ears, eyes, skin
  \item pay follows scarcity: free code is worthless
  \item what a user actually feels --- AI cannot do alone
  \item the key skill: talking about your \textbf{feelings and emotions}
  \end{itemize}
  }

\plush{
  \pptBanner{But Can We Trust Them?}
  \begin{itemize}
  \item same as with humans: tests, review, observability
  \item their knowledge is portable --- just text files
  \item but AI doesn't get defensive, moody, or resentful
  \item it accepts our concerns and just \textbf{fixes, fixes, fixes}
  \end{itemize}
  }

\plush{
  \pptBanner{AI Will Feel Us}
  \begin{itemize}
  \item anything writeable migrates to AI; tech is fully delegated
  \item what it feels like to be a person --- stays ours
  \item tech geeks of the past become humanitarians of the future
  \item the last unexportable skill is \textbf{empathy}
  \end{itemize}
  }
