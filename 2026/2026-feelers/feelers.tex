\plush{
  \pptBanner{Now, When AI Is Uncertain, It Asks Me}
  \begin{itemize}
  \item I tell AI what to do, explaining the intent
  \item it asks me to refine the design, not implementation details
  \item it expects me to fill the \textbf{gaps}
  \end{itemize}
  }

\plush{
  \pptBanner{Today, We Humans Are the Integrators}
  \begin{itemize}
  \item works fine, but responsibility is still ours
  \item we put pieces together and only delegate implementation to AI
  \item AI writes, we hold the architecture --- it \textbf{burns us out}
  \end{itemize}
  }

\plush{
  \pptBanner{Soon, AI Holds the Entire System --- We Hold the Intent}
  \begin{itemize}
  \item AI will know it all: software, data, servers, network, hardware
  \item we won't touch SSH, SQL console, even IDEs or text editors
  \item we will only tell it what we need, in plain \textbf{text}
  \end{itemize}
  }

\plush{
  \pptBanner{The Flip: The Machine Will Ask Us}
  \begin{itemize}
  \item AI starts asking: go there, ask that person, collect the data
  \item the asker becomes the answerer
  \item since for AI, everything is text, we all become \textbf{writers}
  \end{itemize}
  }

\plush{
  \pptBanner{Then, We Become the Senses}
  \begin{itemize}
  \item our real value for AI would be our senses
  \item what a user actually feels --- AI cannot do alone
  \item we'll have value if we can turn what we see into into \textbf{text}
  \end{itemize}
  }

\plush{
  \pptBanner{But Can We Trust Them?}
  \begin{itemize}
  \item we can't trust humans either
  \item however, humans get resentful when we voice it
  \item but AI doesn't get defensive or moody
  \item it gladly accepts our concerns and just \textbf{fixes, fixes, fixes}
  \end{itemize}
  }

\plush{
  \pptBanner{Eventually, AI Will Feel Us}
  \begin{itemize}
  \item AI has no feelings of its own --- that's why it needs ours
  \item what it feels like to be a person --- stays ours
  \item tech geeks of the past become humanitarians of the future
  \item the last unexportable skill is \textbf{empathy}
  \end{itemize}
  }

\plush{
  \pptBanner{For Example}
  \begin{itemize}
  \item AI: "this module is messy --- safe to refactor?"
  \item me: "Betty shipped it Friday --- she's \textbf{proud} of it"
  \item AI waits
  \end{itemize}
  }
